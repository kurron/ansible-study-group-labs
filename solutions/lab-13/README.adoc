== Make a copy of lab-12.yml
----
cp lab-12.yml lab-13.yml
----

== Adjust the debug module to only work when --verbose is applied
----
#!/usr/bin/env ansible-playbook

- name: Exercise Debug Module
  hosts: all
  become: false
  tasks:
      - name: Print Hello
        debug:
            msg: 'Ron was here!'
            verbosity: 1
----
