== Playbook
.lab-22.yml
----
#!/usr/bin/env ansible-playbook
---
- name: Understand Registered Variables
  hosts: all
  become: false
  tasks:
      - name: Determine Login Name
        command: whoami
        register: login
      - debug:
            var: login
            verbosity: 1
      - name: Print Login Name
        debug:
            msg: 'Logged in as user {{ login.stdout }}'
            verbosity: 0
----
