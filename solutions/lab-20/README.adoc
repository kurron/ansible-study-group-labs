== Playbook
.lab-20.yml
----
#!/usr/bin/env ansible-playbook
---
- name: Create Ad Hoc Groups
  hosts: all
  become: false
  tasks:
      - name: Create groups based on distribution
        group_by: key={{ ansible_distribution }}

- name: Print an Ubuntu group message
  hosts: Ubuntu
  become: false
  tasks:
      - name: Print Message
        debug:
            msg: 'Ubuntu rules!'
            verbosity: 0

- name: Print an Amazon group message
  hosts: Amazon
  become: false
  tasks:
      - name: Print Message
        debug:
            msg: 'Amazon rules!'
            verbosity: 0
----
